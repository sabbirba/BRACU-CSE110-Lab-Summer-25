import java.util.Scanner;

public class q1 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number: ");
        int N = sc.nextInt();
        for (int i = 0; i < N; i++) { // loop N times to read pairs of integers
            int X = sc.nextInt();
            int Y = sc.nextInt();
            int sum = 0;
            int count = 0;
            int num = X;
            while (count < Y) { // find sum of the first Y odd numbers starting from X
                if (num % 2 != 0) { // check if the number is odd
                    sum += num;
                    count++; // increment count for each odd number found
                }
                num++; 
            }
            System.out.println(sum);
        }
        sc.close();
    }
}

import java.util.Scanner;

public class q3 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Start: ");
        int start = sc.nextInt();
        System.out.print("End: ");
        int end = sc.nextInt();

        for (int num = start; num <= end; num++) { // loop through the range from start to end
            int isPrime = 1;
            if (num < 2) {
                isPrime = 0;
            } else {
                for (int i = 2; i * i <= num; i++) { // check for factors from 2 to the square root of num
                    if (num % i == 0) {
                        isPrime = 0;
                        break;
                    }
                }
            }
            if (isPrime == 1) { // if the number is prime
                System.out.println(num); 
            }
        }
        sc.close();
    }
}

import java.util.Scanner;

public class q4 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Number of Customer: ");
        int n = sc.nextInt();

        for (int i = 1; i <= n; i++) {
            System.out.print("Item Number for Customer-" + i + ": "); 
            int k = sc.nextInt();
            double sum = 0;
            for (int item = 1; item <= k; item++) { // loop through each item
                System.out.print("Item-" + item + ": ");
                double price = sc.nextDouble();
                sum += price;
            }
            double avg = sum / k;
            System.out.println("Average price for Customer-" + i + ": " + avg); // average price
        } 
        sc.close();
    }
}

import java.util.Scanner;

public class q5 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Start: ");
        int start = sc.nextInt();
        System.out.print("End: ");
        int end = sc.nextInt();

        System.out.println("Armstrong numbers:");

        for (int num = start; num <= end; num++) { // loop through the range
            int temp = num;
            int digits = 0;
           
            for (int t = temp; t > 0; t /= 10) { // count the number of digits
                digits++;
            }
            int sum = 0;
            for (int t = num; t > 0; t /= 10) { // calculate sum of the digits raised to the power of the number of digits
                int d = t % 10;
                int pow = 1;
                for (int i = 0; i < digits; i++) { // raise d to the power of digits
                    pow *= d;
                }
                sum += pow;
            }
            if (sum == num) {
                System.out.println(num);
            }
        }
        sc.close();
    }
}
